
CREATE TABLE areas (
  radius NUMBER(10),
  area NUMBER(14, 4)
);

DECLARE
  v_pi   CONSTANT NUMBER := 3.14159265;
  v_area NUMBER(14, 4);
BEGIN
  DELETE FROM areas;
  
  FOR v_radius IN 5..9 LOOP
    v_area := v_pi * POWER(v_radius, 2);
    
    INSERT INTO areas (radius, area)
    VALUES (v_radius, v_area);
    
    DBMS_OUTPUT.PUT_LINE('Inserted Radius: ' || v_radius || ', Area: ' || v_area);
  END LOOP;
  
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Calculation and insertion complete.');
EXCEPTION
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
    ROLLBACK;
END;
/
